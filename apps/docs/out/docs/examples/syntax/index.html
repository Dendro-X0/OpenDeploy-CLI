Syntax Highlighting Demo

The docs site is configured with rehype-prism-plus and a Prism theme. The following fenced blocks should be highlighted.

TypeScript

ts
export interface User {
  id: string
  email: string
  readonly createdAt: Date
}

export function greet(u: User): string {
  return `Hello, ${u.email}!`
}
